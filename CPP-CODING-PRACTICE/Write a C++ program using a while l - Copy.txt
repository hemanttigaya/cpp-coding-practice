Write a C++ program using a while loop to find the factorial of a given number.

#include<iostream>
using namespace std;
int main(){
    int number;
    cout<<"enter the number:";
    cin>>number;
        long long factorial =1;

    int i =1;
    while(i<=number){
    factorial = factorial*i;
    i++;
}
cout<<"factorial number :"<<factorial;
    
}
