7. *Electricity Bill Calculation:*
   Write a program that calculates the electricity bill using these rules:

   * For units ≤ 100 → ₹1 per unit
   * For 101–200 → ₹2 per unit
   * Above 200 → ₹3 per unit

#include<iostream>
using namespace std ;
int main(){
    int unit;
    float bill;
    cout<<"enter the units :";
    cin>>unit;
    
    if(unit<=100){
    bill=unit*1;
    }
    
    else if(unit<=200){
        bill=(100*1)+(unit-100)*2;
        
    }
    
    else{
        bill=(100*1)+(100*2)+(unit-200)*3;
    }
    cout<<"electricity bill:"<<bill<<endl;
    return 0;
}
