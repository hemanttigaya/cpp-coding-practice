c ++ program to find number of day in months 

#include<iostream>
using namespace std ;
int main(){
    int month ;
    cout<<"enter the months (1-12):";
    cin>>month ;
    
    switch(month){
        case 1: case 3:case 5:case 7:case 8:case 9: case 10:
        cout<<"31 days";
        break ;
        case 4 :case 6:case 11:
        cout<<"30 days";
        break;
        case 2:
        cout<<"28 and 29 days (depends on leap year)";
        break;
        default:
        cout<<"invalid months";
        
    }
    return 0;
}
